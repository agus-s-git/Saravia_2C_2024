78static void tareaAguayAlimento(void *pvParameter){
79 while (true)
80 {
81 if (ON == true)
82 {
83 LedOn(LED_1);
84
85 DISTANCIA = HcSr04ReadDistanceInCentimeters();
86
87 //Del volumen de un cilindro: 2*pi*r^2*altura despejo la altura mínima
88 //y máxima para cada volumen de agua
89
90 //altura para medio litro de agua
91 float h_min = (0.5/(3.14*100)) - 30 ; //cm3 / cm2
92 //altura para 2.5 litro de agua
93 float h_max = (2.5/(3.14*100)) - 30 ;
94
95 float h_faltante = 2.5/(3.14*100) - DISTANCIA ;
96
97 //Le resto al volumen total el volumen faltante y obtengo el volumen que me queda
98 VOLUMEN_AGUA = 2.5 - 2*3.14*100*h_faltante;
99
100 if (DISTANCIA < h_min)
101 {
102 //Activo electrovalvula
103 GPIOOn(GPIO_AGUA);
104 }
105 if (DISTANCIA > h_max)
106 {
107 //Apago electrovalvula
108 GPIOOff(GPIO_AGUA);
109 }
110
111 //Ahora para el alimento
112
113 AnalogInputReadSingle(CH0, &volt_analog);
114
115 //Conversion de mV a gramos
116
117 PESO_ALIMENTO = (volt_analog/*mV*/ * 1000/*gramos*/) / 3300 ;//mV
118
119 if (PESO_ALIMENTO < 50)
120 {
121 GPIOOn(GPIO_ALIMENTO);
122 }
123 if (PESO_ALIMENTO > 500)
124 {
125 GPIOOff(GPIO_ALIMENTO);
126 }
127
128 //Ahora la parte de UART
129
130 UartSendString(UART_PC, "Agua: ");
131 UartSendString(UART_PC, (char*)UartItoa(VOLUMEN_AGUA, 10));
132 UartSendString(UART_PC, "cm3, Alimento: ");
133 UartSendString(UART_PC, (char*)UartItoa(PESO_ALIMENTO, 10));
134 UartSendString(UART_PC, "gr");
135 UartSendString(UART_PC, "\r\n");
136 }
137
138 vTaskDelay(DELAY_MEDIR / portTICK_PERIOD_MS);
139 }
140
141}
